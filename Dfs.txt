def dfs(graph, start, visited=None):
    if visited is None:
        visited = set()
        
    print(start, end=" ")
    visited.add(start)
    
    for neighbour in graph[start]:
        if neighbour not in visited:
            dfs(graph, neighbour, visited)

# Example graph (same as BFS)
graph = {
    'A': ['B', 'C'],
    'B': ['D', 'E'],
    'C': ['F'],
    'D': [],
    'E': ['F'],
    'F': []
}

print("DFS Traversal Order:")
dfs(graph, 'A')
